fn config_lookup(host: &str) -> Result<Option<ForgeType>, GfError> {
    let cfg = match load_config()? {
        Some(c) => c,
        None => return Ok(None),
    };
    Ok(cfg.forge.iter().find(|e| e.domain == host).map(|e| e.forge_type))
}
